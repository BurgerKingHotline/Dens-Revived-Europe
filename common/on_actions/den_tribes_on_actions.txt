on_actions = {

	on_startup = {
		effect = {
			if = {
				limit = {
					has_country_flag = NPC_tribal_country
				}
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = 5000
				}
				if = {
					limit = {
						arms_factory < 2
					}
					random_owned_controlled_state = {
						add_extra_state_shared_building_slots = 1
						add_building_construction = {
							type = arms_factory
							level = 1
							instant_build = yes
						}
					}
					random_owned_controlled_state = {
						add_extra_state_shared_building_slots = 1
						add_building_construction = {
							type = arms_factory
							level = 1
							instant_build = yes
						}
					}
				}
			}
			if = {
				limit = {
					tag = ZAA
				}
				145 = {
					create_unit = {
                				division = "name = \"Tribal Division\" division_template = \"Tribal Division\" start_experience_factor = 0.2"
                				owner = PREV
            				}
				}
			}
		}
	}
	
	on_justifying_wargoal_pulse = {
		effect = {
			if = {
				limit = {
					FROM = { has_idea = ZZZ_tribal_nation_at_peace }
				}
				FROM = {
					swap_ideas = {
						remove_idea = ZZZ_tribal_nation_at_peace 
						add_idea = ZZZ_tribal_nation_at_war
					}
					add_ai_strategy = {
						type = force_build_armies
						value = 50
					}
					country_lock_all_division_template = no
				}
			}
		}
	}

	on_war = {
		effect = {
			if = {
				limit = {
					THIS = { has_idea = ZZZ_tribal_nation_at_peace NOT = { tag = ZAA } }
				}
				THIS = {
					if = {
						limit = {
							num_divisions < 5
						}
						random_owned_controlled_state = {
							limit = {
								any_neighbor_state = {
									NOT = { is_owned_by = PREV }
								}
							}
							create_unit = {
                division = "name = \"Tribal Division\" division_template = \"Tribal Division\" start_experience_factor = 0.2"
                owner = PREV
                count = 2
            	}
						}
						capital_scope = {
							create_unit = {
                division = "name = \"Tribal Division\" division_template = \"Tribal Division\" start_experience_factor = 0.2"
                owner = PREV
                count = 2
            	}
						}
					}
					swap_ideas = {
						remove_idea = ZZZ_tribal_nation_at_peace 
						add_idea = ZZZ_tribal_nation_at_war
					}
					add_ai_strategy = {
						type = force_build_armies
						value = 25
					}
					country_lock_all_division_template = no
				}
			}
		}
	}

	on_peace = {
		effect = {
			if = {
				limit = {
					THIS = {
						has_idea = ZZZ_tribal_nation_at_war
					}
				}
				THIS = {
					swap_ideas = {
						add_idea = ZZZ_tribal_nation_at_peace 
						remove_idea = ZZZ_tribal_nation_at_war
					}
					add_ai_strategy = {
						type = force_build_armies
						value = -25
					}
					country_lock_all_division_template = yes
				}
			}
		}
	}

}
